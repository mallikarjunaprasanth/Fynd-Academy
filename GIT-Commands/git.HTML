step -1   create folder in vs code 

    1. git init                                     -- initialzing repo 

    2. git remote add origin             -- Connecting repo

    3. git branch - checking branch                 -- Checking branch

    4. git checkout -b             -- Create a new branch 


 step-2    modify code 
    
    5. git add .                                    -- All changed files are added

    6. git status                                   -- Checking changed file

    8. git commit -m "commit code"                  -- Commit code

    9. git push origin branch/master 

    10. git merge master branch                     -- Merging master to branch

    11. git pull                                    -- fetch code from branch/master
